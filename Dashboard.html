import { onAuthStateChanged, signOut } from "https://www.gstatic.com/firebasejs/10.7.1/firebase-auth.js"; import { auth } from "./firebase.js"; // ✅ Redirect to login if user not logged in onAuthStateChanged(auth, (user) => { if (!user) { window.location.href = "login.html"; } }); // ✅ Logout button window.logoutUser = function () { signOut(auth) .then(() => { alert("Logged out ✅"); window.location.href = "login.html"; }) .catch(err => alert(err.message)); };
